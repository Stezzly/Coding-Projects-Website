Project Steps:

  1. Download and install Visual Studio Code or CLion.
  2. Ensure you have a C++ compiler installed. For example, you can install GCC or use the built-in compiler in Visual Studio.

  1. Open your chosen IDE (Visual Studio Code or CLion).
  2. Create a new file named student_grade_system.cpp.

  1. Open the student_grade_system.cpp file in your IDE.
  2. Type the following code: 
    
#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Student {
public:
    string name;
    float grade;

    Student(string n, float g) {
        name = n;
        grade = g;
    }
};

void displayGrades(const vector<Student> &students) {
    for (const auto &student : students) {
        cout << student.name << ": " << student.grade << endl;
    }
}

float calculateAverage(const vector<Student> &students) {
    float sum = 0;
    for (const auto &student : students) {
        sum += student.grade;
    }
    return sum / students.size();
}

Student findTopStudent(const vector<Student> &students) {
    Student topStudent = students[0];
    for (const auto &student : students) {
        if (student.grade > topStudent.grade) {
            topStudent = student;
        }
    }
    return topStudent;
}

int main() {
    vector<Student> students;
    string name;
    float grade;
    char choice;

    do {
        cout << "Enter student name: ";
        cin >> name;
        cout << "Enter student grade: ";
        cin >> grade;
        students.emplace_back(name, grade);

        cout << "Do you want to add another student? (y/n): ";
        cin >> choice;
    } while (choice == 'y' || choice == 'Y');

    cout << "\nStudent Grades:\n";
    displayGrades(students);

    float average = calculateAverage(students);
    cout << "\nAverage Grade: " << average << endl;

    Student topStudent = findTopStudent(students);
    cout << "Top Student: " << topStudent.name << " with grade " << topStudent.grade << endl;

    return 0;
}
  3. Save the file.

  1. Open the terminal in your IDE and navigate to the directory containing student_grade_system.cpp.
  2. Type the following command to compile your application: 
    g++ student_grade_system.cpp -o student_grade_system
  3. Type the following command to run your application: 
    ./student_grade_system
  4. Enter student names and grades, and see the average grade and top student.
